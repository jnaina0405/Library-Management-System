#include <iostream>
#include <fstream>
#include <vector>
#include <string>

using namespace std;

class Book {
public:
    int id;
    string title;
    string author;
    bool issued;

    Book(int i, string t, string a, bool is)
        : id(i), title(t), author(a), issued(is) {}
};

vector<Book> library;

void loadFromFile() {
    ifstream file("library.txt");
    if (!file) return;

    int id;
    string title, author;
    bool issued;

    while (file >> id >> issued) {
        file.ignore();
        getline(file, title);
        getline(file, author);
        library.push_back(Book(id, title, author, issued));
    }
    file.close();
}

void saveToFile() {
    ofstream file("library.txt");
    for (auto &b : library) {
        file << b.id << " " << b.issued << endl;
        file << b.title << endl;
        file << b.author << endl;
    }
    file.close();
}

void addBook() {
    int id;
    string title, author;

    cout << "Enter Book ID: ";
    cin >> id;
    cin.ignore();

    cout << "Enter Title: ";
    getline(cin, title);

    cout << "Enter Author: ";
    getline(cin, author);

    library.push_back(Book(id, title, author, false));
    saveToFile();

    cout << "Book added successfully.\n";
}

void viewBooks() {
    if (library.empty()) {
        cout << "No books available.\n";
        return;
    }

    for (auto &b : library) {
        cout << "ID: " << b.id
             << " | Title: " << b.title
             << " | Author: " << b.author
             << " | Status: " << (b.issued ? "Issued" : "Available") << endl;
    }
}

void issueBook() {
    int id;
    cout << "Enter Book ID to issue: ";
    cin >> id;

    for (auto &b : library) {
        if (b.id == id && !b.issued) {
            b.issued = true;
            saveToFile();
            cout << "Book issued successfully.\n";
            return;
        }
    }
    cout << "Book not available.\n";
}

void returnBook() {
    int id;
    cout << "Enter Book ID to return: ";
    cin >> id;

    for (auto &b : library) {
        if (b.id == id && b.issued) {
            b.issued = false;
            saveToFile();
            cout << "Book returned successfully.\n";
            return;
        }
    }
    cout << "Invalid book ID.\n";
}

int main() {
    loadFromFile();
    int choice;

    do {
        cout << "\n--- Library Management System ---\n";
        cout << "1. Add Book\n";
        cout << "2. View Books\n";
        cout << "3. Issue Book\n";
        cout << "4. Return Book\n";
        cout << "5. Exit\n";
        cout << "Enter choice: ";
        cin >> choice;

        switch (choice) {
            case 1: addBook(); break;
            case 2: viewBooks(); break;
            case 3: issueBook(); break;
            case 4: returnBook(); break;
            case 5: cout << "Exiting...\n"; break;
            default: cout << "Invalid choice.\n";
        }
    } while (choice != 5);

    return 0;
}
